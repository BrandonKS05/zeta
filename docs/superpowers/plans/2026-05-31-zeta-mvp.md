# Zeta MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a hackathon/VC-credible Zeta MVP: local `.tex` in → semantic/precheck graph → reviewer report + CLI artifact + CI demo out → optional live backend/LLM enhancement.

**Architecture:** `precheck.js` is the deterministic core (LaTeX parse → notation graph → issues → report). A thin Node.js CLI wraps it for Proof-CI. The popup/content scripts consume the same module for in-browser review. The Python backend provides optional live Lean/LLM/SQLite enhancement but is not required for the demo path.

**Tech Stack:** Node.js (precheck CLI), Chrome Extension (MV3), FastAPI + SQLite (optional backend), Modal (optional GPU inference), GitHub Actions (CI demo)

- [ ] **Step 3: Create `apps/overleaf-extension/precheck/cli.js`**

```js
#!/usr/bin/env node
"use strict";

const fs = require("node:fs");
const path = require("node:path");

// Load precheck engine (UMD module, sets globalThis.__zetaPrecheck)
require("./precheck.js");
const precheck = globalThis.__zetaPrecheck;

function parseArgs(argv) {
  const args = { dir: null, files: [], out: null, json: false, quiet: false };
  for (let i = 2; i < argv.length; i++) {
    if (argv[i] === "--dir" && argv[i + 1]) { args.dir = argv[++i]; }
    else if (argv[i] === "--file" && argv[i + 1]) { args.files.push(argv[++i]); }
    else if (argv[i] === "--out" && argv[i + 1]) { args.out = argv[++i]; }
    else if (argv[i] === "--json") { args.json = true; }
    else if (argv[i] === "--quiet") { args.quiet = true; }
  }
  return args;
}

function collectTexFiles(dir) {
  const results = [];
  for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
    const full = path.join(dir, entry.name);
    if (entry.isDirectory()) {
      results.push(...collectTexFiles(full));
    } else if (entry.isFile() && entry.name.endsWith(".tex")) {
      results.push(full);
    }
  }
  return results;
}

function main() {
  const args = parseArgs(process.argv);

  let filePaths = [...args.files];
  if (args.dir) {
    if (!fs.existsSync(args.dir)) {
      console.error(`zeta: directory not found: ${args.dir}`);
      process.exit(1);
    }
    filePaths.push(...collectTexFiles(args.dir));
  }

  if (filePaths.length === 0) {
    console.error("zeta: no .tex files found. Use --dir <path> or --file <path.tex>");
    process.exit(1);
  }

  const files = filePaths.map((fp) => ({
    file_path: args.dir ? path.relative(args.dir, fp) : path.basename(fp),
    content: fs.readFileSync(fp, "utf8"),
  }));

  const report = precheck.buildPrecheckReport(files);

  const outDir = args.out || process.cwd();

  const jsonOut = path.join(outDir, "zeta-report.json");
  const mdOut = path.join(outDir, "zeta-report.md");

  const jsonPayload = {
    score: report.score,
    certification: report.certification,
    issues: (report.analysis?.issues || []).map((issue) => ({
      type: issue.type,
      symbol: issue.symbol,
      severity: issue.severity,
      title: precheck.issueTitle ? precheck.issueTitle(issue) : issue.type,
      location: precheck.issueLocation ? precheck.issueLocation(issue) : null,
    })),
    reviewLedger: report.reviewLedger || [],
    files: files.map((f) => f.file_path),
    generatedAt: new Date().toISOString(),
  };

  fs.writeFileSync(jsonOut, JSON.stringify(jsonPayload, null, 2));
  fs.writeFileSync(mdOut, precheck.markdownReviewerReport(report));

  if (!args.quiet) {
    console.log(`Zeta Pre-Check Complete`);
    console.log(`  Files scanned : ${files.length}`);
    console.log(`  Issues found  : ${jsonPayload.issues.length}`);
    console.log(`  Readiness     : ${report.score}/100`);
    console.log(`  Certification : ${report.certification?.label || "—"}`);
    console.log(`  JSON report   : ${jsonOut}`);
    console.log(`  MD report     : ${mdOut}`);
  }

  const hasHighSeverity = jsonPayload.issues.some((i) => i.severity === "high" || i.severity === "error");
  process.exit(hasHighSeverity ? 1 : 0);
}

main();
```
